值:
do \
{ \
t _heap_p = (p)[(i)]; \
size_t _heap_i = (i); \
size_t _heap_j = (i); \
while ((_heap_i = (_heap_i << 1U) + 1U) < (n)) \
{ \
if (_heap_i != (n)-1U && \
func((p)[_heap_i], (p)[_heap_i + 1U])) \
{ \
++_heap_i; \
} \
if (func((p)[_heap_i], _heap_p)) \
{ \
break; \
} \
(p)[_heap_j] = (p)[_heap_i]; \
_heap_j = _heap_i; \
} \
(p)[_heap_j] = _heap_p; \
} while (0)
